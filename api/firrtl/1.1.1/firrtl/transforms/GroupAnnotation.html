case class GroupAnnotation(components: Seq[ComponentName], newModule: String, newInstance: String, outputSuffix: Option[String] = None, inputSuffix: Option[String] = None) extends Annotation with Product with Serializable

Specifies a group of components, within a module, to pull out into their own module Components that are only connected to a group's components will also be included
